## Quick Start Examples

### Working with NFTs (ERC-721)

```javascript
const { ERC721Fetcher } = require('big-world-bigger-ideas');

// Initialize with contract address
const nftFetcher = new ERC721Fetcher('0xBC4CA0EdA7647A8aB7C2061c2E118A18a936f13D');

// Get NFT owner
const owner = await nftFetcher.getOwner('1');
console.log('Owner:', owner);

// Get token URI
const tokenURI = await nftFetcher.getTokenURI('1');
console.log('Token URI:', tokenURI);

// Get contract name
const name = await nftFetcher.getName();
console.log('Collection:', name);
```

### Bitcoin Mining Data

```javascript
const { BitcoinMiningFetcher } = require('big-world-bigger-ideas');

// Initialize fetcher
const btcFetcher = new BitcoinMiningFetcher();

// Get hash rate for the past week
const hashRate = await btcFetcher.getHashRate('1w');
console.log('Hash Rate:', hashRate);

// Get difficulty adjustment data
const difficulty = await btcFetcher.getDifficulty('1m');
console.log('Difficulty:', difficulty);
```

### Multi-chain Blockchain Data

```javascript
const { BlockchairFetcher } = require('big-world-bigger-ideas');

// Initialize for Bitcoin
const btcBlockchair = new BlockchairFetcher('bitcoin');

// Get block data
const block = await btcBlockchair.getBlock(700000);
console.log('Block:', block);

// Get address data
const address = await btcBlockchair.getAddress('1A1zP1eP5QGefi2DMPTfTL5SLmv7DivfNa');
console.log('Address:', address);
```

### Token UUID Generation

```javascript
const { TokenUUID } = require('big-world-bigger-ideas');

// Generate UUID v4
const uuid4 = TokenUUID.generateV4();
console.log('UUID v4:', uuid4);

// Generate UUID v5 (deterministic)
const uuid5 = TokenUUID.generateV5('token-data', TokenUUID.NAMESPACE_URL);
console.log('UUID v5:', uuid5);
```

### Feature Flags Management

```javascript
const { FeatureFlags } = require('big-world-bigger-ideas');

// Initialize feature flags
const flags = new FeatureFlags('./feature-flags.json');

// Check if a feature is enabled
const isEnabled = flags.isEnabled('new-feature');

// Enable a feature
flags.setFlag('new-feature', true);

// List all flags
const allFlags = flags.listFlags();
console.log('All flags:', allFlags);
```

## Running Tests

The package includes comprehensive tests for all modules:

```bash
# Run all tests
npm test

# Run specific module tests
npm run test:erc721
npm run test:bitcoin-mining
npm run test:feature-flags
```

## Running Examples

Each module includes a demo script:

```bash
# Run ERC-721 demo
npm run erc721:demo

# Run Bitcoin mining demo
npm run bitcoin-mining:demo

# Run feature flags demo
npm run feature-flags:demo
```
